2) Criar um programa que lê uma lista(array) e apresenta apenas os números impares, o conteúdo da lista deve ser [5, 8, 7, 9, 2, 5, 3]:

Resposta:

function question2() {
  const numbers = [5, 8, 7, 9, 2, 5, 3];
  let oddNumbers = [];
  for (let i = 0; i < numbers.length; i++) {
    const positionValue = numbers[i];
      if (positionValue % 2 !== 0) {
      oddNumbers.push(positionValue);
    }
  }
  return oddNumbers;
}

  var n = array.length;
  for (var i = 0; i < array.length; i++) {
    for (var j = 1; j < n - i; j++) {
      if (array[j - 1] > array[j]) {

function question4(num = 20) {
  var arr = [0, 1];
  for (var i = 2; i < num; i++) {
    arr.push(arr[i - 2] + arr[i - 1]);
  }
